A {peca_movida.cor} comeu a peça {p_adversaria.id} do {j.cor}!") p_adversaria.na_base = True p_adversaria.posicao_relativa = -1 p_adversaria.posicao_global = -1 def mover_peca(self, jogador, peca, passos): # 1. Verificar se o movimento ultrapassa o tabuleiro if peca.posicao_relativa + passos > self.limite_passos: peca.completou = True peca.posicao_global = -1 print(f"🎉 Peça {peca.id} chegou ao fim e está segura!") return True # 3. Verificar Bloqueios no caminho # Para simplificar, checamos apenas se a casa de *destino* está bloqueada nova_pos_relativa = peca.posicao_relativa + passos nova_pos_global = (nova_pos_relativa + peca.offset) % 52 if self.verificar_bloqueios(nova_pos_global, jogador.cor): print(f"Movimento inválido! A casa {nova_pos_global} possui um BLOQUEIO adversário.") return False # 4. Efetuar o movimento peca.atualizar_posicao_global(passos) print(f"Peça {peca.id} moveu para a casa global {peca.posicao_global}.") # 5. Checar Captura self.checar_e_capturar(jogador, peca) return True def iniciar_rodada(self): jogador_atual = self.jogadores[self.turno_atual] print(f"\n--- Turno: {jogador_atual.cor} ---") jogar_novamente = True while jogar_novamente: input(f"[{jogador_atual.cor}] Pressione Enter para rolar o dado...") dado = self.rolar_dado() print(f"🎲 Dado: {dado}") # Filtrar jogadas válidas pecas_disponiveis = [] for p in ...: # Filtro de peças if not p.completou: if p.na_base and dado == 6: pecas_disponiveis.append(p) elif not p.na_base: pecas_disponiveis.append(p) if not pecas_disponiveis: print("Sem jogadas válidas.") jogar_novamente = False # Passa a vez se não puder mover else: print("Escolha uma peça:") for i, p in enumerate(pecas_disponiveis): status = "Na Base" if p.na_base else f"Casa Global {p.posicao_global}" print(f"[{i}] Peça {p.id} ({status})") escolha = -1 while escolha < 0 or escolha >= len(pecas_disponiveis): try: escolha = int(input("Número da peça: ")) except ValueError: pass movimentos_validos = self.mover_peca(jogador_atual, pecas_disponiveis[escolha], dado) # Regra do 6: Se tirou 6 (e fez um movimento válido), joga de novo! if dado == 6 and movimentos_validos: print("🔥 Você tirou um 6!
